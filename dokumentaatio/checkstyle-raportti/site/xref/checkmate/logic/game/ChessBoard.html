4   import checkmate.logic.pieces.King;

10  import java.util.ArrayList;
11  
12  /**
13   * Class for the chessboard.
14   *
15   * @author llmlks
16   */
17  public class ChessBoard {
18  
19      /**
20       * ArrayList for type Square.
21       */
22      private final ArrayList<Square> squares;
23      /**
24       * ArrayList for type Piece.
25       */
26      private final ArrayList<Piece> pieces;
27      /**
28       * Integer value for size of one side of the board (8 * 8 squares).
29       */
30      private final int size = 8;
31  
32      /**
33       * Constructor generates empty ArrayLists for pieces and squares.
34       */
35      public ChessBoard() {
36          this.squares = new ArrayList<>();
37          this.pieces = new ArrayList<>();
38      }
39  
40      /**
41       * Returns private variable squares.
42       *
43       * @return List of squares on the board
44       */
45      public final ArrayList<Square> getSquares() {
46          return this.squares;
47      }
48  
49      /**
50       * Returns private variable pieces.
51       *
52       * @return List of pieces on the board
53       */
54      public final ArrayList<Piece> getPieces() {
55          return this.pieces;
56      }
57  
58      /**
59       * Initialises squares; creates 64 squares and adds them to this board's
60       * list of squares.
61       */
62      public final void initSquares() {
63          for (int i = 1; i <= size; i++) {
64              for (int j = 1; j <= size; j++) {
65                  Square square = new Square(i, j);
66                  square.setSize(size);
67                  squares.add(square);
68              }
69          }
70      }
71  
72      /**
73       * Calls appropriate methods to initialise pieces, i.e. initPawns(),
74       * initPiecesExclPawns() and setPiecesToSquares().
75       */
76      public final void initPieces() {
77          initPawns();
78          initPiecesExclPawns();
79          setPiecesToSquares();
80      }
81  
82      /**
83       * Initialises pawns; creates 12 of both colours with appropriate initial
84       * squares and adds them to pieces list.
85       */
86      public final void initPawns() {
87          int[] rows = new int[]{2, size - 1};
88          for (int row : rows) {
89              String colour = "w";
90              if (row == 2) {
91                  colour = "b";
92              }
93              for (int i = 1; i <= size; i++) {
94                  pieces.add(new Pawn(new Square(i, row), colour));
95              }
96          }
97      }
98  
99      /**
100      * Initialises pieces other than pawns, creating 12 of each colour with
101      * appropriate initial squares, and adding them to pieces list.
102      */
103     public final void initPiecesExclPawns() {
104         int[] rows = new int[]{1, size};
105         String[] initials = new String[]{"rook", "knight", "bishop", "queen",
106             "king", "bishop", "knight", "rook"};
107         for (int row : rows) {
108             String colour = "w";
109             if (row == 1) {
110                 colour = "b";
111             }
112             for (int j = 0; j < size; j++) {
113                 switch (initials[j]) {
114                     case "rook":
115                         pieces.add(new Rook(new Square(j + 1, row), colour));
116                         break;
117                     case "knight":
118                         pieces.add(new Knight(new Square(j + 1, row), colour));
119                         break;
120                     case "bishop":
121                         pieces.add(new Bishop(new Square(j + 1, row), colour));
122                         break;
123                     case "queen":
124                         pieces.add(new Queen(new Square(j + 1, row), colour));
125                         break;
126                     default:
127                         pieces.add(new King(new Square(j + 1, row), colour));
128                         break;
129                 }
130             }
131         }
132     }
133 
134     /**
135      * Matches all the pieces on board to appropriate squares and vice versa.
136      */
137     public final void setPiecesToSquares() {
138         for (Piece p : this.pieces) {
139             for (Square s : this.squares) {
140                 if (p.getSquare().equals(s) && p.getAvailable()) {
141                     s.setPiece(p);
142                     p.setSquare(s);
143                 }
144             }
145         }
146     }
147 
148     /**
149      * Promotes pawn to another piece of the player's choice.
150      *
151      * @param p Pawn to be promoted
152      * @param promo Piece to be promoted to
153      * @param player Player whose piece is promoted
154      */
155     public final void promote(Piece p, String promo, Player player) {
156         switch (promo) {
157             case "rook":
158                 Rook r = new Rook(p.getSquare(), p.getColour());
159                 player.addPiece(r);
160                 this.pieces.add(r);
161                 break;
162             case "knight":
163                 Knight k = new Knight(p.getSquare(), p.getColour());
164                 player.addPiece(k);
165                 this.pieces.add(k);
166                 break;
167             case "bishop":
168                 Bishop b = new Bishop(p.getSquare(), p.getColour());
169                 player.addPiece(b);
170                 this.pieces.add(b);
171                 break;
172             default:
173                 Queen q = new Queen(p.getSquare(), p.getColour());
174                 player.addPiece(q);
175                 this.pieces.add(q);
176                 break;
177         }
178         player.removePiece(p);
179         this.pieces.remove(p);
180         setPiecesToSquares();
181     }
182 
183     /**
184      * Finds square at coordinates x and y.
185      *
186      * @param x integer x coordinate
187      * @param y integer y coordinate
188      * @return Square at coordinates x and y, or null if none is found
189      */
190     public final Square findSquareByCoordinates(int x, int y) {
191         for (Square s : this.getSquares()) {
192             if (s.getX() == x && s.getY() == y) {
193                 return s;
194             }
195         }
196         return null;
197     }
198 }

10  public class Knight extends Piece {
11  
12      /**
13       * Calls constructor of parent class Piece and sets this.type to "knight".
14       *
15       * @param s Square
16       * @param c String
17       */
18      public Knight(final Square s, final String c) {
19          super(s, c);
20          this.setType("knight");
21      }
22  
23      /**
24       * Checks whether the move to square s from this.square is valid for knight.
25       *
26       * @param s Square
27       * @return boolean
28       */
29      @Override
30      public final boolean isValidMove(final Square s) {
31          Square square = this.getSquare();
32          if (Math.abs(s.getX() - square.getX()) == 2 && Math.abs(s.getY()
33                  - square.getY()) == 1) {
34              return true;
35          }
36          return Math.abs(s.getX() - square.getX()) == 1 && Math.abs(s.getY()
37                  - square.getY()) == 2;
38      }
39  
40      @Override
41      public final ArrayList<Square> possibleMoves() {
42          ArrayList<Square> possible = new ArrayList<>();
43          return possible;
44      }

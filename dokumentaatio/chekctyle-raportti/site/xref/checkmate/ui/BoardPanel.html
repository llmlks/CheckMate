37  public class BoardPanel extends JPanel implements MouseListener {
38  
39      private final ChessGame chess;
40      private final HashMap<Piece, String> pieces;
41      private Piece chosen;
42      private ArrayList<Square> possibleMoves;
43  
44      public BoardPanel(ChessGame game) {
45          this.chess = game;
46          this.pieces = initPieces();
47      }
48  
49      @Override
50      public void mouseClicked(MouseEvent e) {
51          int x = e.getX() / 75;
52          int y = e.getY() / 75;
53          Square clicked = this.chess.findSquareByCoordinates(y + 1, x + 1);
54          Component c = this.findComponentAt(e.getX(), e.getY());
55          if (chosen != null && possibleMoves.contains(clicked)) {
56              chosen.move(clicked);
57              chosen = null;
58              repaint();
59          } else {
60              if (c instanceof JLabel && clicked.isOccupied()) {
61                  this.chosen = clicked.getPiece();
62  
63                  switch (this.chosen.getType()) {
64                      case "pawn":
65                          chosen = (Pawn) chosen;
66                          break;
67                      case "rook":
68                          chosen = (Rook) chosen;
69                          break;
70                      case "knight":
71                          chosen = (Knight) chosen;
72                          break;
73                      case "queen":
74                          chosen = (Queen) chosen;
75                          break;
76                      case "king":
77                          chosen = (King) chosen;
78  
79                  }
80                  possibleMoves = this.chess.getValidator().validMoves(chosen);
81                  Container parent = c.getParent();
82                  parent.setBackground(new Color(255, 121, 77));
83              }
84          }
85      }

104     public void createComponents() {
105         for (Square square : this.chess.getBoard().getSquares()) {
106             int i = square.getX() - 1;
107             int j = square.getY() - 1;
108             JPanel sqr = new JPanel();
109             sqr.setSize(75, 75);
110             sqr.setBackground(new Color(209, 139, 71));
111             if ((i + j) % 2 == 0) {
112                 sqr.setBackground(new Color(255, 206, 158));
113             }
114             if (square.isOccupied()) {
115                 try {
116                     JLabel label = new JLabel(new ImageIcon(ImageIO.read(new File(pieces.get(square.getPiece())))));
117                     sqr.add(label);
118                 } catch (IOException ex) {
119                     JLabel label = new JLabel("Missing file.");
120                     sqr.add(label);
121                 }
122             }
123             this.add(sqr);
124         }
125 
126     }
127 
128     private HashMap<Piece, String> initPieces() {
129         HashMap<Piece, String> pieceIcons = new HashMap<>();
130         for (Piece piece : this.chess.getBoard().getPieces()) {
131             String icon = "icons/" + piece.getType() + "_" + piece.getColour() + ".png";
132             pieceIcons.put(piece, icon);
133         }
134         return pieceIcons;
135     }
